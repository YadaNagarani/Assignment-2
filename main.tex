\section{Question No 2.15}
Find the equation of the line passing through \myvec{-3 \\ 5} and perpendicular to the line through the points \myvec{2 \\ 5} and \myvec{-3 \\ 6}
%
\section{SOLUTION} 
\begin{align}
\Vec{A} =  \myvec{2 \\ 5 } , \Vec{B} =  \myvec{-3 \\ 6}
\\
\text{If the direction vector is \myvec{1 \\ m} the slope is m}
\end{align}
\begin{align}
\Vec{m} = \myvec{-3 \\ 6}-\myvec{2 \\ 5}
\\
\Vec{m} = \myvec{-5 \\ 1}
\\
\Vec{m} = \frac{-1}{5}
\end{align}
Given point $\vec{P} = \myvec{-3\\5}$ and slope $m = \frac{-1}{5}$.
\begin{align}
\label{eq:line_norm_dir}
\vec{n} &= \myvec{0&-1\\1&0}\vec{m} 
\\
\vec{n} &= \myvec{0&-1\\1&0}\myvec{\frac{-1}{5}}
\\
&= \myvec{\frac{1}{5} \\ \frac{-1}{5}}
\end{align}
The equation of the line in terms of the normal vector is then obtained as
\begin{align}
\label{eq:line_norm_vec}
\vec{n}^T\brak{\vec{x}-\vec{P}} &= 0
\\
\implies \myvec{1 & 5} \vec{x} &= 22
\end{align}
